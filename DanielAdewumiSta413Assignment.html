# Scatter and Line Plots
plt.figure(figsize=(15, 5))
for year in ["2017(X1)", "2018(X2)", "2019(X3)"]:
    plt.scatter(df["Month(Y)"], df[year], label=f'Scatter Y vs X{year[-1]}', alpha=0.7)
    plt.plot(df["Month(Y)"], df[year], label=f'Line Y vs X{year[-1]}')
plt.xlabel("Month")
plt.ylabel("Values")
plt.title("Scatter and Line Plot of Y with X1, X2, X3")
plt.xticks(rotation=45)
plt.legend()
plt.tight_layout()
